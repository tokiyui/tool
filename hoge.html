最新NCファイル取得（自動ファイル名抽出）

import js, re, asyncio from pyodide.http import pyfetch import datetime PROXY = "https://worker.ryohorita105.workers.dev/?url=" BASE = "https://data.marine.copernicus.eu/product/WAVE_GLO_PHY_SWH_L3_NRT_014_001/files?path=WAVE_GLO_PHY_SWH_L3_NRT_014_001/global_vavh_l3_rt_cfo_" async def fetch_nc_py(): now = datetime.datetime.utcnow() minus6 = now - datetime.timedelta(hours=6) hour = (minus6.hour // 3) * 3 latest = datetime.datetime(minus6.year, minus6.month, minus6.day, hour) YYYYMMDDTHHMMSS = latest.strftime("%Y%m%dT%H%M%S") dirUrl = f"{PROXY}{BASE}{latest.year:04d}/{latest.month:02d}/{latest.day:02d}/" js.document.getElementById("status").innerText += f"\n[PyScript] ディレクトリURL: {dirUrl}\n取得中..." try: resp = await pyfetch(dirUrl) if resp.status != 200: raise RuntimeError(f"HTTP error {resp.status}") html = await resp.string() regex = re.compile(f"global_vavh_l3_rt_cfo_{YYYYMMDDTHHMMSS}_[^\"']+\\.nc") matches = regex.findall(html) if not matches: raise RuntimeError("ファイルが見つかりません") latestFile = matches[0] js.document.getElementById("status").innerText += f"\n[PyScript] 最新ファイル: {latestFile}\n取得中..." fileResp = await pyfetch(f"{PROXY}{BASE}{latest.year:04d}/{latest.month:02d}/{latest.day:02d}/{latestFile}") if fileResp.status != 200: raise RuntimeError(f"HTTP error {fileResp.status}") data = await fileResp.bytes() js.document.getElementById("status").innerText += f"\n[PyScript] 取得成功: {len(data)} bytes" except Exception as e: js.document.getElementById("status").innerText += f"\n[PyScript] 取得失敗: {e}" async def setup_button(): btn = js.document.createElement("button") btn.innerText = "PyScriptで最新ファイル取得" js.document.body.appendChild(btn) from pyodide.ffi import create_proxy btn.addEventListener("click", create_proxy(lambda e: asyncio.ensure_future(fetch_nc_py()))) asyncio.ensure_future(setup_button())
