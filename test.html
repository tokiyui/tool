// 1. 地図の初期化（日本近海を中心に設定） const map = L.map('map').setView([30, 138], 4); // 背景地図（無地に近いシンプルなダークマップ） L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', { attribution: '© OpenStreetMap © CARTO' }).addTo(map); // 2. ユーザー指定の色分けロジック const getColor = h => { if (h === null || h === undefined) return "transparent"; // データがない場合 if (h < 0.5) return "dodgerblue"; if (h < 1.0) return "limegreen"; if (h < 1.5) return "greenyellow"; if (h < 2.0) return "gold"; if (h < 3.0) return "darkorange"; return "orangered"; }; // 3. CORS回避用プロキシとNOAA ERDDAP API URL const proxy = "https://worker.ryohorita105.workers.dev/?url="; const targetUrl = "https://coastwatch.pfeg.noaa.gov/erddap/griddap/noaacwNrtJ3S3aS3bS6aSwhMerged.geoJson?sign_wave_height[(last)][(0.0):1:(-0.0)][(-80.0):1:(80.0)][(0.0):1:(360.0)]"; // プロキシURLとターゲットURLを結合してリクエスト fetch(proxy + encodeURIComponent(targetUrl)) .then(response => { if (!response.ok) throw new Error('ネットワークエラーまたはプロキシエラーが発生しました'); return response.json(); }) .then(data => { // GeoJSONとして地図に追加 L.geoJSON(data, { pointToLayer: function (feature, latlng) { const h = feature.properties.sign_wave_height; const markerColor = getColor(h); return L.circleMarker(latlng, { radius: 4, fillColor: markerColor, color: markerColor, weight: 1, opacity: 0.8, fillOpacity: 0.7 }).bindPopup(`有義波高: ${h.toFixed(2)} m`); } }).addTo(map); }) .catch(error => { console.error("データ取得エラー:", error); alert("データの取得に失敗しました。プロキシまたはデータソースの一時的な問題の可能性があります。"); }); // 4. 画面右下に凡例（カラーバー）を追加 const legend = L.control({position: 'bottomright'}); legend.onAdd = function (map) { const div = L.DomUtil.create('div', 'info legend'); const grades = [0, 0.5, 1.0, 1.5, 2.0, 3.0]; const labels = ['波高 (m)
']; for (let i = 0; i < grades.length; i++) { const from = grades[i]; const to = grades[i + 1]; labels.push( ' ' + from + (to ? '–' + to : '+') ); } div.innerHTML = labels.join('
'); return div; }; legend.addTo(map);
